const express =require('express');
const { buildSchema } = require('graphql');
const { graphqlHTTP } = require('express-graphql');

const app = express();

const schema = buildSchema(`
        type Query{
            hello:strig
        }`)

const root={
    hello:()=>{
       return 'hello world'; 
    },
};

app.use('/graphql',graphqlHTTP({
    graphql: true,
    schema :schema ,
    rootValue: root ,

}) );

app.listen(400,() => console.log('server on port 4000'));
